employees:columns like emp_id,emp_name,salary,dept_name

find the employee whose salary is greater than 50000

select emp_name from employees where salary>50000

find the employees who belongs to departments where salary>60000

select emp_name from employees where dept_name in(
select dept_name from employees where salary>60000);


find the employees earning more than the average salary

select emp_name,salary from employees
where salary>(select avg(salary) from employees)

find the average salary of the employees

select emp_name ,(select avg(salary) as avgsal from employees) from employees

ON users2
AFTER UPDATE
AS
BEGIN
    SET NOCOUNT ON;

    UPDATE u
    SET updated_at = SYSDATETIME()
    FROM users2 u

CREATE TABLE users_after (
    id INT PRIMARY KEY,
    name VARCHAR(50),
    email VARCHAR(100)
);

CREATE TABLE user_log (
    log_id INT IDENTITY(1,1) PRIMARY KEY,
    message VARCHAR(200)
);



CREATE TRIGGER trg_after_insert
ON users_after
AFTER INSERT
AS
BEGIN
    INSERT INTO user_log(message)
    SELECT 'User inserted with ID = ' + CAST(id AS VARCHAR)
    FROM inserted;
END;

CREATE TABLE users_before (
    id INT PRIMARY KEY,
    name VARCHAR(50),
    email VARCHAR(100)
);


CREATE TRIGGER trg_instead_insert
ON users_before
INSTEAD OF INSERT
AS
BEGIN
    PRINT 'Before inserting data';

    INSERT INTO users_before (id,name,email)
    SELECT id,name,email
    FROM inserted;
END;

INSERT INTO users_before VALUES (3,'Alice','alice@email.com');

select * from users_before;



CREATE TABLE users_mail (
    id INT PRIMARY KEY,
    name VARCHAR(50),
    email VARCHAR(100)
);


CREATE TRIGGER trg_before_insert_email
ON users_mail
INSTEAD OF INSERT
AS
BEGIN
    IF EXISTS (
        SELECT 1 
        FROM users_mail u
        JOIN inserted i ON u.email = i.email
    )
    BEGIN
        PRINT 'Email already exists. Insert not allowed.';
    END
    ELSE
    BEGIN
        INSERT INTO users_mail(id, name, email)
        SELECT id, name, email
        FROM inserted;
    END
END;


CREATE TRIGGER trg_after_insert_email
ON users_mail
AFTER INSERT
AS
BEGIN
    PRINT 'New user with email inserted successfully';
END;
